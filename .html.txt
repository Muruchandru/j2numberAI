<script>
const STORAGE_KEY = 'j2_mcboy_predictor_v2';

function loadData() {
  const def = { history: [], topEndings: ['07','15','58','72','94'] };
  try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || def; } catch { return def; }
}
function saveData(d) { localStorage.setItem(STORAGE_KEY, JSON.stringify(d)); }

function showToast(msg, type='') {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.className = 'toast ' + type;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), 3000);
}

function updateTopEndings(history) {
  const last6 = [...history].reverse().slice(0, 6);
  const freq = {};
  for (const entry of last6)
    for (const num of entry.numbers)
      if (num && num.length >= 2) {
        const e = num.slice(-2);
        freq[e] = (freq[e] || 0) + 1;
      }
  const sorted = Object.entries(freq).sort((a,b) => b[1]-a[1]);
  return sorted.length ? sorted.slice(0,5).map(x=>x[0]) : ['07','15','58','72','94'];
}

function getRecentWinners(history) {
  return [...history].reverse().slice(0,6)
    .flatMap(e => (e.numbers || []).slice(0,3))
    .filter(w => w && w.length === 4);
}

function hybridEnding(e1, e2) { return (e1?.[0]||'0') + (e2?.[1]||'7'); }

function pick(arr) { return arr[Math.floor(Math.random() * arr.length)]; }

function generatePredictions(history, topEndings) {
  const winners = getRecentWinners(history);
  if (!winners.length || !topEndings.length) return { pm3: [], pm9: [] };

  const pm3 = new Set(), pm9 = new Set();
  for (let i = 0; i < 10 && pm3.size < 3; i++)
    pm3.add(pick(winners).slice(0,2) + pick(topEndings));
  for (let i = 0; i < 10 && pm9.size < 3; i++) {
    const prefix = Math.floor(Math.random()*100).toString().padStart(2,'0');
    pm9.add(prefix + hybridEnding(pick(topEndings), pick(topEndings)));
  }
  return { pm3: [...pm3].slice(0,3), pm9: [...pm9].slice(0,3) };
}

function renderPredictions(history, topEndings) {
  const area = document.getElementById('predictionsArea');
  if (history.length < 4) {
    area.innerHTML = `<div class="empty-state"><div class="icon">??</div>Add at least 3 days (6 draws) for reliable predictions.</div>`;
    return;
  }
  const p = generatePredictions(history, topEndings);
  if (!p.pm3.length) {
    area.innerHTML = `<div class="empty-state"><div class="icon">??</div>Not enough data yet. Keep adding results.</div>`;
    return;
  }

  const numHtml = nums => nums.map((n,i) =>
    `<div class="pred-num" style="animation-delay:${i*0.08}s">${n}</div>`
  ).join('');

  area.innerHTML = `
    <div class="pred-block">
      <div class="pred-label">?? 3PM Draw — <span>Next</span></div>
      <div class="numbers-grid">${numHtml(p.pm3)}</div>
    </div>
    <div class="pred-block">
      <div class="pred-label">?? 9PM Draw — <span>Next</span></div>
      <div class="numbers-grid">${numHtml(p.pm9)}</div>
    </div>
    <div style="margin-top:16px;">
      <div class="pred-label" style="margin-bottom:10px;">?? Top endings — last 3 days</div>
      <div class="endings-row">
        ${topEndings.map(e => `<span class="ending-chip">${e}</span>`).join('')}
      </div>
    </div>
    <div class="budget-note">?? Suggested budget: $6 ($1 per number) · Stop-loss: $100</div>
  `;
}

function renderHistory(history) {
  const el = document.getElementById('historyList');
  if (!history.length) {
    el.innerHTML = `<div class="empty-state"><div class="icon">??</div>No data yet.</div>`;
    return;
  }
  el.innerHTML = [...history].reverse().slice(0,10).map(e => `
    <div class="history-item">
      <div>
        <div class="history-date">${e.date}</div>
        <div class="history-type">${e.type === '3pm' ? '?? 3PM' : '?? 9PM'}</div>
      </div>
      <div class="history-nums">${e.numbers.slice(0,8).join(' · ')}${e.numbers.length > 8 ? ' …' : ''}</div>
    </div>
  `).join('');
}

function refreshAll() {
  const data = loadData();
  data.topEndings = updateTopEndings(data.history);
  saveData(data);
  renderHistory(data.history);
  renderPredictions(data.history, data.topEndings);
}

function addResult() {
  const date = document.getElementById('drawDate').value;
  const drawType = document.getElementById('drawType').value;
  const raw = document.getElementById('rawNumbers').value;

  if (!date) { showToast('?? Please select a date', 'error'); return; }
  if (!raw.trim()) { showToast('?? Please enter winning numbers', 'error'); return; }

  const numbers = raw.split(/[\s,\n]+/).map(s=>s.trim())
    .filter(s => s.length === 4 && /^\d{4}$/.test(s));

  if (!numbers.length) { showToast('? No valid 4-digit numbers found', 'error'); return; }

  const data = loadData();
  if (data.history.some(e => e.date === date && e.type === drawType)) {
    showToast(`?? ${date} ${drawType} already saved`, 'error');
    return;
  }

  data.history.push({ id: Date.now(), date, type: drawType, numbers });
  data.history.sort((a,b) => new Date(a.date) - new Date(b.date));
  data.topEndings = updateTopEndings(data.history);
  saveData(data);

  document.getElementById('rawNumbers').value = '';
  document.getElementById('drawDate').value = '';
  refreshAll();
  showToast(`? Saved ${numbers.length} numbers for ${date} ${drawType}`, 'success');
}

function clearAll() {
  if (confirm('?? Delete ALL saved results and reset the app?')) {
    localStorage.removeItem(STORAGE_KEY);
    refreshAll();
    showToast('All data cleared.', '');
  }
}

document.addEventListener('DOMContentLoaded', () => {
  refreshAll();
  document.getElementById('saveBtn').addEventListener('click', addResult);
  document.getElementById('clearHistoryBtn').addEventListener('click', clearAll);
  document.getElementById('refreshBtn').addEventListener('click', refreshAll);
});
